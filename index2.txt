        function processApplication(applicationId, action, comment = '') {
            const application = applications.find(a => a.id === applicationId);
            if (!application) return false;

            application.status = action; // 'approved' or 'rejected'
            application.moderatorComment = comment;
            application.processedAt = new Date();

            if (action === 'approved') {
                // Create campaign from approved application
                const newCampaign = {
                    id: campaigns.length + 1,
                    title: application.title,
                    description: application.description,
                    category: application.category,
                    targetAmount: application.amount,
                    raisedAmount: 0,
                    donorsCount: 0,
                    daysLeft: 30,
                    image: application.image || `https://via.placeholder.com/400x250/3b82f6/ffffff?text=${encodeURIComponent(application.title)}`,
                    status: 'active',
                    createdAt: new Date(),
                    organizer: application.applicantName,
                    organizerId: application.applicantId
                };
                
                campaigns.push(newCampaign);
                renderCampaigns();
                updateStats();
            }

            saveToLocalStorage();

            // Notify applicant
            const applicant = users.find(u => u.id === application.applicantId);
            if (applicant) {
                const statusText = action === 'approved' ? 'tasdiqlandi' : 'rad etildi';
                const message = `Arizangiz ${statusText}. ${comment ? 'Izoh: ' + comment : ''}`;
                
                // Simulate SMS and web notification
                setTimeout(() => {
                    if (currentUser && currentUser.id === applicant.id) {
                        showNotification(
                            `Ariza ${statusText}`, 
                            message, 
                            action === 'approved' ? 'success' : 'warning'
                        );
                    }
                }, 1000);
            }

            return true;
        }

        function processDonation(amount, paymentMethod) {
            if (!currentUser) {
                showNotification('Xatolik', 'Xayriya qilish uchun tizimga kiring', 'error');

            }

            if (!currentDonationCampaign || !amount || !paymentMethod) {
                showNotification('Xatolik', 'Barcha maydonlarni to\'ldiring', 'error');
                return false;
            }

            // Create payment record
            const newPayment = {
                id: payments.length + 1,
                campaignId: currentDonationCampaign.id,
                campaignTitle: currentDonationCampaign.title,
                donorName: `${currentUser.firstName} ${currentUser.lastName}`,
                donorId: currentUser.id,
                amount: amount,
                paymentMethod: paymentMethod,
                status: 'completed',
                paidAt: new Date()
            };

            payments.push(newPayment);

            // Update campaign
            currentDonationCampaign.raisedAmount += amount;
            currentDonationCampaign.donorsCount += 1;

            saveToLocalStorage();
            renderCampaigns();
            updateStats();
            hideAllModals();

            showNotification('Rahmat!', `${formatCurrency(amount)} xayriya muvaffaqiyatli amalga oshirildi`, 'success');

            // Reset donation state
            selectedDonationAmount = 0;
            selectedPaymentMethod = '';
            currentDonationCampaign = null;

            return true;

            return campaigns.filter(campaign =>

        function updateStats() {
            const totalRaised = campaigns.reduce((sum, c) => sum + c.raisedAmount, 0);
            const totalDonors = campaigns.reduce((sum, c) => sum + c.donorsCount, 0);
            const totalHelped = campaigns.filter(c => c.raisedAmount >= c.targetAmount).length;

            // Animate counters
            animateCounter(elements.totalCampaigns, campaigns.length);
            animateCounter(elements.totalRaised, totalRaised, true);
            animateCounter(elements.totalDonors, totalDonors);
            animateCounter(elements.totalHelped, totalHelped);
        }

        function animateCounter(element, target, isCurrency = false) {
            const start = 0;
            const duration = 2000;
            const startTime = performance.now();

            function update(currentTime) {
                const elapsed = currentTime - startTime;
                const progress = Math.min(elapsed / duration, 1);
                const current = Math.floor(start + (target - start) * progress);
                
                if (isCurrency) {
                    element.textContent = formatCurrency(current);
                } else {
                    element.textContent = formatNumber(current);
                }

                if (progress < 1) {
                    requestAnimationFrame(update);
                }
            }

            requestAnimationFrame(update);
        }
